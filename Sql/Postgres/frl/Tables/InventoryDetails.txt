CREATE TABLE IF NOT EXISTS frl."InventoryDetails"
(
    "Id"                uuid                            NOT NULL
,   "RowCreated"        timestamp(3) with time zone     NOT NULL    default CURRENT_TIMESTAMP(3)
,   "RowModified"       timestamp(3) with time zone     NOT NULL    default CURRENT_TIMESTAMP(3)
,   "InventoryId"       uuid                            NOT NULL
,   "Name"              character varying(250)          NOT NULL
,   "Value"             character varying(250)          NOT NULL
,   "Description"       character varying(1000)             NULL

,   CONSTRAINT "PK_InventoryDetails" PRIMARY KEY ("Id")
,   CONSTRAINT "FK_InventoryDetails_InventoryId" FOREIGN KEY ("InventoryId")
        REFERENCES frl."Inventory" ("Id") MATCH SIMPLE
        ON UPDATE NO ACTION
        ON DELETE CASCADE
);
