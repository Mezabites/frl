CREATE TABLE IF NOT EXISTS frl."ApiSyncResult"
(
    "Id"                uuid                            NOT NULL
,   "RowCreated"        timestamp(3) with time zone     NOT NULL    DEFAULT CURRENT_TIMESTAMP(3)
,   "RowModified"       timestamp(3) with time zone     NOT NULL    DEFAULT CURRENT_TIMESTAMP(3)
,   "ApiSyncId"         uuid                            NOT NULL
,   "LastSyncId"        uuid                                NULL
,   "ObjectType"        character varying(256)          NOT NULL
,   "ObjectName"        character varying(1000)         NOT NULL
,   "Action"            smallint                        NOT NULL
,   "Error"             character varying(1000)             NULL

,   CONSTRAINT "PK_ApiSyncResult" PRIMARY KEY ("Id")
,   CONSTRAINT "FK_ApiSyncResult_ApiSyncId" FOREIGN KEY ("ApiSyncId")
        REFERENCES frl."ApiSync" ("Id") MATCH SIMPLE
        ON UPDATE NO ACTION
        ON DELETE CASCADE
);
