CREATE TABLE IF NOT EXISTS frl."ProjectTaskEntryUser" 
(
    "Id"                    uuid                            NOT NULL
,   "RowCreated"            timestamp(3) with time zone     NOT NULL    DEFAULT CURRENT_TIMESTAMP(3)
,   "RowModified"           timestamp(3) with time zone     NOT NULL    DEFAULT CURRENT_TIMESTAMP(3)
,   "ProjectTaskEntryId"    uuid                            NOT NULL
,   "DateTime"              timestamp(3) with time zone     NOT NULL
,   "UserId"                character varying(255)          NOT NULL
,   "Units"                 integer                         NOT NULL    DEFAULT 1
,   "Days"                  integer                         NOT NULL    DEFAULT 1
,   "Hours"                 integer                             NULL
,   "WorkerPayment"         decimal                         NOT NULL
,   "Notes"                 character varying(1000)             NULL

,   CONSTRAINT "PK_ProjectTaskEntryUser" PRIMARY KEY ("Id")
,   CONSTRAINT "FK_ProjectTaskEntryUser_ProjectTaskEntryId" FOREIGN KEY ("ProjectTaskEntryId")
        REFERENCES frl."ProjectTaskEntry" ("Id") MATCH SIMPLE
        ON UPDATE NO ACTION
        ON DELETE NO ACTION
);
