CREATE TABLE IF NOT EXISTS frl."ProjectReportTask"
(
    "Id"                uuid                            NOT NULL
,   "RowCreated"        timestamp(3) with time zone     NOT NULL    DEFAULT CURRENT_TIMESTAMP(3)
,   "ProjectId"         uuid                            NOT NULL
,   "ProjectTaskId"     uuid                            NOT NULL

,   CONSTRAINT "PK_ProjectReportTask" PRIMARY KEY ("Id")
,   CONSTRAINT "FK_ProjectReportTask_ProjectId" FOREIGN KEY ("ProjectId")
        REFERENCES frl."Project" ("Id") MATCH SIMPLE
        ON UPDATE NO ACTION
        ON DELETE CASCADE
,   CONSTRAINT "FK_ProjectReportTask_ProjectTaskId" FOREIGN KEY ("ProjectTaskId")
        REFERENCES frl."ProjectTask" ("Id") MATCH SIMPLE
        ON UPDATE NO ACTION
        ON DELETE CASCADE
);
