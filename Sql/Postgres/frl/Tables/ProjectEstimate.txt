DROP TABLE frl."ProjectEstimate" CASCADE; 

CREATE TABLE IF NOT EXISTS frl."ProjectEstimate" 
(
    "Id"                uuid                            NOT NULL
,   "RowCreated"        timestamp(3) with time zone     NOT NULL    DEFAULT CURRENT_TIMESTAMP(3)
,   "RowModified"       timestamp(3) with time zone     NOT NULL    DEFAULT CURRENT_TIMESTAMP(3)
,   "ProjectId"         uuid                            NOT NULL
,   "Name"              character varying(256)          NOT NULL
,   "Number"            character varying(25)           NOT NULL
,   "StatusClvId"       uuid                            NOT NULL
,   "ClientContactId"   uuid                                NULL
,   "Notes"             character varying(1000)             NULL
,   "TaskColumns"       varchar(1000)                   NOT NULL
,   "EntryColumns"      varchar(1000)                       NULL

,   CONSTRAINT "PK_ProjectEstimate" PRIMARY KEY ("Id")
,   CONSTRAINT "FK_ProjectEstimate_ProjectId" FOREIGN KEY ("ProjectId")
        REFERENCES frl."Project" ("Id") MATCH SIMPLE
        ON UPDATE NO ACTION
        ON DELETE CASCADE
,   CONSTRAINT "FK_ProjectEstimate_StatusClvId" FOREIGN KEY ("StatusClvId")
        REFERENCES app."ClassifierValue" ("Id") MATCH SIMPLE
        ON UPDATE NO ACTION
        ON DELETE NO ACTION
,   CONSTRAINT "FK_Project_ClientContactId" FOREIGN KEY ("ClientContactId")
        REFERENCES frl."ClientContact" ("Id") MATCH SIMPLE
        ON UPDATE NO ACTION
        ON DELETE NO ACTION
);
